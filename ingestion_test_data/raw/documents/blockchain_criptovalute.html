/** * @title LendingPool - Protocollo di Lending Decentralizzato * @dev Implementa un pool di lending con interessi composti */ contract LendingPool is ERC20, ReentrancyGuard, Ownable { struct Deposit { uint256 amount; uint256 timestamp; uint256 interestRate; } struct Loan { uint256 amount; uint256 collateral; uint256 timestamp; uint256 interestRate; bool liquidated; } mapping(address => Deposit) public deposits; mapping(address => Loan) public loans; uint256 public totalDeposits; uint256 public totalLoans; uint256 public baseInterestRate = 5; // 5% APY base uint256 public utilizationRate; // Percentuale di utilizzo del pool event DepositMade(address indexed user, uint256 amount); event WithdrawalMade(address indexed user, uint256 amount, uint256 interest); event LoanTaken(address indexed user, uint256 amount, uint256 collateral); event LoanRepaid(address indexed user, uint256 amount, uint256 interest); event Liquidation(address indexed user, uint256 collateralSeized); /** * @dev Calcola tasso di interesse basato sull'utilizzazione */ function calculateInterestRate() public view returns (uint256) { // Modello di interesse: base + (utilization * multiplier) uint256 utilization = totalLoans * 100 / (totalDeposits + 1); return baseInterestRate + (utilization * 10 / 100); } /** * @dev Deposita ETH nel pool di lending */ function deposit() external payable nonReentrant { require(msg.value > 0, "Deposit amount must be positive"); uint256 interestRate = calculateInterestRate(); if (deposits[msg.sender].amount > 0) { // Calcola interessi maturati uint256 timeElapsed = block.timestamp - deposits[msg.sender].timestamp; uint256 interestEarned = deposits[msg.sender].amount * interestRate * timeElapsed / (365 days * 100); deposits[msg.sender].amount += interestEarned; } deposits[msg.sender].amount += msg.value; deposits[msg.sender].timestamp = block.timestamp; deposits[msg.sender].interestRate = interestRate; totalDeposits += msg.value; // Mint LP tokens proporzionali al deposito uint256 lpTokens = msg.value * totalSupply() / (totalDeposits - msg.value + 1); if (totalSupply() == 0) { lpTokens = msg.value; } _mint(msg.sender, lpTokens); emit DepositMade(msg.sender, msg.value); } /** * @dev Preleva deposito con interessi */ function withdraw(uint256 lpAmount) external nonReentrant { require(balanceOf(msg.sender) >= lpAmount, "Insufficient LP tokens"); uint256 depositShare = lpAmount * totalDeposits / totalSupply(); require(depositShare <= deposits[msg.sender].amount, "Insufficient deposit balance" ); // Calcola interessi maturati uint256 timeElapsed=block.timestamp - deposits[msg.sender].timestamp; uint256 interestEarned=depositShare * deposits[msg.sender].interestRate * timeElapsed / (365 days * 100); uint256 totalWithdrawal=depositShare + interestEarned; deposits[msg.sender].amount -=depositShare; totalDeposits -=depositShare; _burn(msg.sender, lpAmount); payable(msg.sender).transfer(totalWithdrawal); emit WithdrawalMade(msg.sender, depositShare, interestEarned); } /** * @dev Richiede un prestito con collateral */ function borrow(uint256 borrowAmount) external payable nonReentrant { require(msg.value> 0, "Collateral required"); require(borrowAmount <= totalDeposits - totalLoans, "Insufficient liquidity" ); uint256 collateralRatio=msg.value * 100 / borrowAmount; require(collateralRatio>= 150, "Insufficient collateral (min 150%)"); uint256 interestRate = calculateInterestRate(); loans[msg.sender] = Loan({ amount: borrowAmount, collateral: msg.value, timestamp: block.timestamp, interestRate: interestRate, liquidated: false }); totalLoans += borrowAmount; payable(msg.sender).transfer(borrowAmount); emit LoanTaken(msg.sender, borrowAmount, msg.value); } /** * @dev Restituisce prestito con interessi */ function repay() external payable nonReentrant { require(loans[msg.sender].amount > 0, "No active loan"); require(!loans[msg.sender].liquidated, "Loan already liquidated"); uint256 timeElapsed = block.timestamp - loans[msg.sender].timestamp; uint256 interestDue = loans[msg.sender].amount * loans[msg.sender].interestRate * timeElapsed / (365 days * 100); uint256 totalDue = loans[msg.sender].amount + interestDue; require(msg.value >= totalDue, "Insufficient repayment amount"); totalLoans -= loans[msg.sender].amount; // Restituisci collateral payable(msg.sender).transfer(loans[msg.sender].collateral); // Restituisci eccesso if (msg.value > totalDue) { payable(msg.sender).transfer(msg.value - totalDue); } delete loans[msg.sender]; emit LoanRepaid(msg.sender, loans[msg.sender].amount, interestDue); } /** * @dev Liquida posizione underwater */ function liquidate(address borrower) external nonReentrant { require(loans[borrower].amount > 0, "No active loan"); require(!loans[borrower].liquidated, "Already liquidated"); uint256 currentCollateralValue = loans[borrower].collateral; // In produzione userebbe price feed uint256 loanValue = loans[borrower].amount; uint256 collateralRatio = currentCollateralValue * 100 / loanValue; require(collateralRatio < 120, "Position not underwater (min 120% maintenance)" ); // Liquidatore riceve bonus del 5% uint256 liquidationBonus=currentCollateralValue * 5 / 100; uint256 totalSeized=currentCollateralValue + liquidationBonus; loans[borrower].liquidated=true; totalLoans -=loans[borrower].amount; payable(msg.sender).transfer(totalSeized); emit Liquidation(borrower, totalSeized); } /** * @dev View function per informazioni del pool */ function getPoolInfo() external view returns ( uint256 _totalDeposits, uint256 _totalLoans, uint256 _utilizationRate, uint256 _interestRate ) { _totalDeposits=totalDeposits; _totalLoans=totalLoans; _utilizationRate=totalLoans * 100 / (totalDeposits + 1); _interestRate=calculateInterestRate(); } /** * @dev View function per posizione utente */ function getUserPosition(address user) external view returns ( uint256 depositAmount, uint256 loanAmount, uint256 collateralAmount, uint256 depositInterestRate, uint256 loanInterestRate ) { depositAmount=deposits[user].amount; loanAmount=loans[user].amount; collateralAmount=loans[user].collateral; depositInterestRate=deposits[user].interestRate; loanInterestRate=loans[user].interestRate; } }

3.1 Automated Market Makers (AMM)

import numpy as np from typing import Tuple, List class AutomatedMarketMaker: """ Implementazione di Automated Market Maker (AMM) simile a Uniswap V2 """ def __init__(self, token_a_reserve: float, token_b_reserve: float, fee: float = 0.003): """ Inizializza AMM pool Args: token_a_reserve: Riserva iniziale token A token_b_reserve: Riserva iniziale token B fee: Fee di trading (0.3% default) """ self.token_a_reserve = token_a_reserve self.token_b_reserve = token_b_reserve self.fee = fee self.k = token_a_reserve * token_b_reserve # Costante prodotto AMM self.total_liquidity_tokens = np.sqrt(token_a_reserve * token_b_reserve) def get_amount_out(self, amount_in: float, reserve_in: float, reserve_out: float) -> float: """ Calcola amount out usando formula AMM con fee Formula: amount_out = (amount_in * (1 - fee) * reserve_out) / (reserve_in + amount_in * (1 - fee)) """ amount_in_with_fee = amount_in * (1 - self.fee) numerator = amount_in_with_fee * reserve_out denominator = reserve_in + amount_in_with_fee return numerator / denominator def swap_a_for_b(self, amount_a_in: float) -> Tuple[float, float, float]: """ Swap token A per token B Returns: (amount_b_out, new_reserve_a, new_reserve_b) """ amount_b_out = self.get_amount_out(amount_a_in, self.token_a_reserve, self.token_b_reserve) new_reserve_a = self.token_a_reserve + amount_a_in new_reserve_b = self.token_b_reserve - amount_b_out # Verifica che k sia mantenuto (approssimativamente) new_k = new_reserve_a * new_reserve_b assert abs(new_k - self.k) / self.k < 0.01, "Invariant violated" return amount_b_out, new_reserve_a, new_reserve_b def swap_b_for_a(self, amount_b_in: float) -> Tuple[float, float, float]: """ Swap token B per token A Returns: (amount_a_out, new_reserve_a, new_reserve_b) """ amount_a_out = self.get_amount_out(amount_b_in, self.token_b_reserve, self.token_a_reserve) new_reserve_b = self.token_b_reserve + amount_b_in new_reserve_a = self.token_a_reserve - amount_a_out return amount_a_out, new_reserve_a, new_reserve_b def add_liquidity(self, amount_a: float, amount_b: float) -> float: """ Aggiungi liquidità al pool Returns: liquidity_tokens: Token di liquidità mintati """ # Calcola quanto liquidità aggiungere mantenendo ratio if self.token_a_reserve == 0 or self.token_b_reserve == 0: # Pool vuoto liquidity_tokens = np.sqrt(amount_a * amount_b) else: # Calcola basato sul token più limitante liquidity_a = amount_a * self.total_liquidity_tokens / self.token_a_reserve liquidity_b = amount_b * self.total_liquidity_tokens / self.token_b_reserve liquidity_tokens = min(liquidity_a, liquidity_b) # Aggiorna riserve self.token_a_reserve += amount_a self.token_b_reserve += amount_b self.total_liquidity_tokens += liquidity_tokens self.k = self.token_a_reserve * self.token_b_reserve return liquidity_tokens def remove_liquidity(self, liquidity_tokens: float) -> Tuple[float, float]: """ Rimuovi liquidità dal pool Returns: (amount_a_out, amount_b_out) """ # Calcola quote amount_a_out = liquidity_tokens * self.token_a_reserve / self.total_liquidity_tokens amount_b_out = liquidity_tokens * self.token_b_reserve / self.total_liquidity_tokens # Aggiorna riserve self.token_a_reserve -= amount_a_out self.token_b_reserve -= amount_b_out self.total_liquidity_tokens -= liquidity_tokens self.k = self.token_a_reserve * self.token_b_reserve return amount_a_out, amount_b_out def get_price(self) -> float: """ Restituisce prezzo spot A in termini di B """ return self.token_b_reserve / self.token_a_reserve def get_slippage(self, trade_size: float, is_a_for_b: bool = True) -> float: """ Calcola slippage per una data dimensione trade """ if is_a_for_b: amount_out = self.get_amount_out(trade_size, self.token_a_reserve, self.token_b_reserve) price_impact = amount_out / trade_size spot_price = self.get_price() execution_price = amount_out / trade_size slippage = (execution_price - spot_price) / spot_price else: amount_out = self.get_amount_out(trade_size, self.token_b_reserve, self.token_a_reserve) spot_price = 1 / self.get_price() execution_price = amount_out / trade_size slippage = (execution_price - spot_price) / spot_price return slippage def impermanent_loss(self, price_ratio_change: float) -> float: """ Calcola impermanent loss per un cambio nel price ratio Args: price_ratio_change: Nuovo prezzo / vecchio prezzo """ r = price_ratio_change impermanent_loss = 2 * np.sqrt(r) / (1 + r) - 1 return impermanent_loss # Esempio di utilizzo amm = AutomatedMarketMaker(token_a_reserve=100000, token_b_reserve=100000) # Swap amount_out, new_a, new_b = amm.swap_a_for_b(1000) print(f"Swapped 1000 A for {amount_out:.2f} B") # Calcola slippage slippage = amm.get_slippage(1000) print(f"Slippage: {slippage:.4%}") # Aggiungi liquidità liq_tokens = amm.add_liquidity(10000, 10000) print(f"Added liquidity, received {liq_tokens:.2f} LP tokens") # Calcola impermanent loss il = amm.impermanent_loss(1.2) # 20% price increase print(f"Impermanent loss: {il:.4%}")

4. Central Bank Digital Currency (CBDC)

Le CBDC rappresentano l'evoluzione delle valute centrali in formato digitale, combinando i vantaggi della moneta elettronica con il controllo sovrano delle banche centrali.

✅ Vantaggi delle CBDC

  • Efficienza nei Pagamenti: Transazioni instantanee 24/7
  • Riduzione Costi: Eliminazione intermediari tradizionali
  • Inclusione Finanziaria: Accesso universale ai servizi finanziari
  • Programmability: Moneta programmabile con condizioni
  • Anti-Money Laundering: Tracking completo delle transazioni

❌ Sfide delle CBDC

  • Privacy: Rischio sorveglianza di massa
  • Disintermediazione: Impatto sui depositi bancari
  • Cybersecurity: Target di attacchi sofisticati
  • Interoperabilità: Coordinamento internazionale complesso
  • Financial Stability: Rischi sistemici da digitalizzazione

5. Security e Regulatory Considerations

La sicurezza e la regolamentazione rappresentano aspetti critici per l'adozione mainstream delle tecnologie blockchain e criptovalute.

Case Study: The DAO Hack (2016)

Contesto: The DAO era un fondo di investimento decentralizzato su Ethereum che raccolse $150M in crowdfunding.

Vulnerabilità: Reentrancy attack che permise il drain di $50M dai contratti smart.

Lezione Appresa: Necessità di audit di sicurezza approfonditi e best practices di programmazione.

Impatto: Hard fork di Ethereum per recuperare i fondi, creazione di Ethereum Classic.

Implementare multifactor authentication per tutti gli accessi wallet
Utilizzare hardware security modules (HSM) per chiavi crittografiche
Condurre audit di sicurezza regolari con firme specializzate
Implementare circuit breakers per fermare trading in condizioni estreme
Sviluppare piani di disaster recovery e business continuity
Monitorare continuamente per vulnerabilità zero-day
Educare utenti su best practices di sicurezza
Collaborare con autorità regolamentari per compliance
