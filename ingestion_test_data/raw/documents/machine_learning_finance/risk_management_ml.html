Risk Management Avanzato con Machine Learning

Implementazione di modelli predittivi per la gestione del rischio finanziario nell'era del big data

1. Introduzione al Risk Management Quantitativo

Il risk management moderno si basa su approcci data-driven che sfruttano algoritmi di machine learning per prevedere, misurare e mitigare i rischi finanziari. Questa guida approfondisce le metodologie più avanzate per l'implementazione di sistemi di risk management basati su ML.

Contesto Industriale: Le istituzioni finanziarie processano petabytes di dati giornalmente, richiedendo approcci ML scalabili per identificare pattern di rischio complessi e correlazioni non lineari.

2. Modelli Predittivi per Credit Risk

I modelli di credit risk rappresentano una delle applicazioni più mature del ML in finanza, con accuracy che supera significativamente i modelli tradizionali basati su score creditizi.

3. Market Risk e Value at Risk (VaR)

Il calcolo del Value at Risk rappresenta una sfida classica per gli algoritmi ML, che devono bilanciare accuratezza predittiva con robustezza computazionale.

3.1 Conditional VaR con Extreme Value Theory

Expected Shortfall (ES) Formula:
ESα = E[-R | -R > VaRα] = -E[R | R < -VaRα]
from scipy.stats import genextreme, norm from arch import arch_model import numpy as np import pandas as pd class AdvancedVaRModel: def __init__(self, confidence_level=0.95, distribution='historical'): self.confidence_level = confidence_level self.distribution = distribution self.models = {} def calculate_historical_var(self, returns, window=252): """Historical simulation VaR""" if len(returns) < window: window=len(returns) # Rolling window VaR var_series=[] for i in range(window, len(returns)): window_returns=returns[i-window:i] var=-np.percentile(window_returns, (1 - self.confidence_level) * 100) var_series.append(var) return pd.Series(var_series, index=returns.index[window:]) def calculate_parametric_var(self, returns, distribution='normal' ): """Parametric VaR con diverse distribuzioni""" mu=returns.mean() sigma=returns.std() if distribution=='normal' : z_score=norm.ppf(1 - self.confidence_level) var=-(mu + sigma * z_score) elif distribution=='t-student' : from scipy.stats import t # Fit t-distribution params=t.fit(returns) t_score=t.ppf(1 - self.confidence_level, df=params[0]) var=-(mu + sigma * t_score) elif distribution=='ged' : from scipy.stats import genextreme # Fit Generalized Extreme Value distribution params=genextreme.fit(-returns) # Fit to losses gev_score=genextreme.ppf(1 - self.confidence_level, *params) var=gev_score return var def calculate_garch_var(self, returns, model_type='GARCH' ): """GARCH-based VaR""" # Fit GARCH model if model_type=='GARCH' : model=arch_model(returns, vol='Garch' , p=1, q=1, dist='skewt' ) elif model_type=='EGARCH' : model=arch_model(returns, vol='EGarch' , p=1, q=1, dist='skewt' ) else: model=arch_model(returns, vol='Garch' , p=1, q=1, dist='normal' ) fitted_model=model.fit(disp='off' , show_warning=False) # Forecast volatility forecast=fitted_model.forecast(horizon=1) vol_forecast=np.sqrt(forecast.variance.iloc[-1]) # VaR calculation if fitted_model.distribution.name=='skewt' : # Skewed t-distribution parameters mu, sigma, skew, df=fitted_model.params['mu'], vol_forecast.iloc[0], \ fitted_model.params['eta'], fitted_model.params['df'] from scipy.stats import skewnorm # Approximation for skewed t z_score=skewnorm.ppf(1 - self.confidence_level, skew) else: z_score=norm.ppf(1 - self.confidence_level) var=-(fitted_model.params['mu'] + vol_forecast.iloc[0] * z_score) return var, vol_forecast.iloc[0], fitted_model def calculate_expected_shortfall(self, returns, var_value=None): """Expected Shortfall (Conditional VaR)""" if var_value is None: var_value=self.calculate_parametric_var(returns) # ES is average of losses beyond VaR tail_losses=returns[returns < -var_value] if len(tail_losses)==0: # Fallback to parametric approximation es=var_value * (1 / (1 - self.confidence_level)) else: es=-tail_losses.mean() return es def backtest_var(self, returns, var_series, alpha=0.05): """Backtesting VaR model con diversi test statistici""" # Hit series (1 if loss> VaR, 0 otherwise) hits = (returns < -var_series).astype(int) # Unconditional coverage test (Kupiec) n=len(hits) x=hits.sum() # Number of violations p_hat=x / n # Observed violation rate p=1 - self.confidence_level # Expected violation rate # Likelihood ratio test if x> 0 and x < n: lr_uc=-2 * np.log(((1-p)**(n-x) * p**x) / ((1-p_hat)**(n-x) * p_hat**x)) else: lr_uc=np.inf if x==n else 0 # Conditional coverage test (Christoffersen) # Independence test transitions=0 for i in range(1, len(hits)): if hits[i-1]==1 and hits[i]==1: transitions +=1 n00=(hits==0).sum() - (hits==1).sum() if hits.iloc[0]==0 else (hits==0).sum() n01=hits.iloc[0] if hits.iloc[0]==1 else 0 n10=1 - hits.iloc[0] if hits.iloc[0]==0 else 0 n11=transitions if n01 + n11> 0 and n00 + n10 > 0: p01 = n01 / (n00 + n01) p11 = n11 / (n10 + n11) pi_hat = (n01 + n11) / n lr_ind = -2 * np.log( ((1-pi_hat)**(n00+n10) * pi_hat**(n01+n11)) / ((1-p01)**(n00) * p01**n01 * (1-p11)**(n10) * p11**n11) ) else: lr_ind = 0 lr_cc = lr_uc + lr_ind # Results backtest_results = { 'violations': x, 'expected_violations': p * n, 'violation_rate': p_hat, 'expected_rate': p, 'kupiec_test_stat': lr_uc, 'kupiec_p_value': 1 - chi2.cdf(lr_uc, 1), 'christoffersen_test_stat': lr_cc, 'christoffersen_p_value': 1 - chi2.cdf(lr_cc, 2), 'model_rejects_null': lr_uc > chi2.ppf(1-alpha, 1) or lr_cc > chi2.ppf(1-alpha, 2) } return backtest_results def stress_test_var(self, returns, stress_scenarios): """Stress testing VaR sotto diversi scenari""" stress_results = {} for scenario_name, scenario_returns in stress_scenarios.items(): # Calculate VaR under stress scenario var_stress = self.calculate_parametric_var(scenario_returns) es_stress = self.calculate_expected_shortfall(scenario_returns, var_stress) stress_results[scenario_name] = { 'var': var_stress, 'expected_shortfall': es_stress, 'max_loss': -scenario_returns.min(), 'worst_percentile': -np.percentile(scenario_returns, 1) } return stress_results # Usage example var_model = AdvancedVaRModel(confidence_level=0.99) # Historical VaR historical_var = var_model.calculate_historical_var(portfolio_returns) # Parametric VaR parametric_var = var_model.calculate_parametric_var(portfolio_returns, 't-student') # GARCH VaR garch_var, vol, model = var_model.calculate_garch_var(portfolio_returns) # Expected Shortfall es = var_model.calculate_expected_shortfall(portfolio_returns) # Backtesting backtest_results = var_model.backtest_var(portfolio_returns, historical_var) print(f"Historical VaR (99%): {historical_var.iloc[-1]:.4f}") print(f"Parametric VaR (99%): {parametric_var:.4f}") print(f"GARCH VaR (99%): {garch_var:.4f}") print(f"Expected Shortfall (99%): {es:.4f}") print(f"Backtest Violations: {backtest_results['violations']} (expected: {backtest_results['expected_violations']:.1f})")

4.1 Anomaly Detection con Autoencoders

from tensorflow.keras.models import Model from tensorflow.keras.layers import Input, Dense, Dropout from sklearn.preprocessing import StandardScaler, MinMaxScaler from sklearn.metrics import classification_report, roc_auc_score import numpy as np class FraudAutoencoder: def __init__(self, input_dim, encoding_dim=32): self.input_dim = input_dim self.encoding_dim = encoding_dim self.autoencoder = None self.encoder = None self.scaler = StandardScaler() self.threshold = None def build_autoencoder(self): """Costruzione autoencoder per anomaly detection""" input_layer = Input(shape=(self.input_dim,)) # Encoder encoded = Dense(128, activation='relu')(input_layer) encoded = Dropout(0.2)(encoded) encoded = Dense(64, activation='relu')(encoded) encoded = Dropout(0.2)(encoded) encoded = Dense(self.encoding_dim, activation='relu')(encoded) # Decoder decoded = Dense(64, activation='relu')(encoded) decoded = Dropout(0.2)(decoded) decoded = Dense(128, activation='relu')(decoded) decoded = Dropout(0.2)(decoded) decoded = Dense(self.input_dim, activation='sigmoid')(decoded) # Models self.autoencoder = Model(input_layer, decoded) self.encoder = Model(input_layer, encoded) self.autoencoder.compile(optimizer='adam', loss='mse') def train_autoencoder(self, normal_data, epochs=100, batch_size=64): """Training su dati normali (unsupervised)""" # Scale data scaled_data = self.scaler.fit_transform(normal_data) # Train history = self.autoencoder.fit( scaled_data, scaled_data, epochs=epochs, batch_size=batch_size, validation_split=0.2, shuffle=True, verbose=0 ) return history def set_threshold(self, normal_data, percentile=95): """Imposta threshold basato su reconstruction error dei dati normali""" scaled_data = self.scaler.transform(normal_data) reconstructions = self.autoencoder.predict(scaled_data, verbose=0) # Mean squared error per sample mse = np.mean(np.power(scaled_data - reconstructions, 2), axis=1) # Threshold at specified percentile self.threshold = np.percentile(mse, percentile) return self.threshold def detect_anomalies(self, data, return_scores=False): """Rileva anomalie nei nuovi dati""" scaled_data = self.scaler.transform(data) reconstructions = self.autoencoder.predict(scaled_data, verbose=0) # Reconstruction errors mse_scores = np.mean(np.power(scaled_data - (cm[0,1] * fp_cost) print(f"Financial Impact: ${financial_impact:,.0f}") # Precision at different thresholds thresholds = np.linspace(self.threshold * 0.5, self.threshold * 2, 20) precision_scores = [] for thresh in thresholds: preds_at_thresh = (scores > thresh).astype(int) if preds_at_thresh.sum() > 0: precision = (y_true & preds_at_thresh).sum() / preds_at_thresh.sum() else: precision = 1.0 precision_scores.append(precision) return { 'auc': auc, 'confusion_matrix': cm, 'financial_impact': financial_impact, 'precision_at_thresholds': dict(zip(thresholds, precision_scores)) } # Usage example # Assume transaction data with features transaction_features = extract_transaction_features(transactions) # Split normal vs fraudulent transactions normal_transactions = transaction_features[transaction_features['is_fraud'] == 0] fraud_transactions = transaction_features[transaction_features['is_fraud'] == 1] # Remove label from features feature_cols = [col for col in transaction_features.columns if col != 'is_fraud'] normal_data = normal_transactions[feature_cols] fraud_data = fraud_transactions[feature_cols] # Initialize and train autoencoder fraud_detector = FraudAutoencoder(input_dim=len(feature_cols)) fraud_detector.build_autoencoder() fraud_detector.train_autoencoder(normal_data) # Set threshold threshold = fraud_detector.set_threshold(normal_data) print(f"Anomaly Threshold: {threshold:.6f}") # Evaluate evaluation = fraud_detector.evaluate_fraud_detection( normal_data.sample(1000), # Test normal fraud_data.sample(200) # Test fraud )

5. Liquidity Risk e Network Analysis

Il liquidity risk richiede approcci avanzati che considerano le interconnessioni tra istituzioni finanziarie e la propagazione dei rischi attraverso la rete finanziaria.

5.1 Graph Neural Networks per Systemic Risk
