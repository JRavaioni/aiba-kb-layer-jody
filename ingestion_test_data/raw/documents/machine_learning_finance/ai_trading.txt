### 1. Analisi Tecnica con Machine Learning

#### 1.1 Pattern Recognition con Convolutional Neural Networks

Le reti neural convoluzionali (CNN) hanno dimostrato eccellenti capacità nel riconoscimento di pattern tecnici nei grafici dei prezzi. A differenza dei tradizionali indicatori tecnici, le CNN possono identificare pattern complessi e non lineari che sfuggono all'analisi umana.

**Implementazione Pratica:**

```python
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Conv2D, MaxPooling2D, Flatten, Dense, Dropout
import numpy as np
import pandas as pd

class TechnicalPatternCNN:
    def __init__(self, image_height=64, image_width=64, num_classes=3):
        self.image_height = image_height
        self.image_width = image_width
        self.num_classes = num_classes
        self.model = self._build_model()

    def _build_model(self):
        model = Sequential([
            # Convolutional layers for pattern recognition
            Conv2D(32, (3, 3), activation='relu', input_shape=(self.image_height, self.image_width, 1)),
            MaxPooling2D((2, 2)),
            Conv2D(64, (3, 3), activation='relu'),
            MaxPooling2D((2, 2)),
            Conv2D(128, (3, 3), activation='relu'),
            MaxPooling2D((2, 2)),

            # Dense layers for classification
            Flatten(),
            Dense(512, activation='relu'),
            Dropout(0.5),
            Dense(256, activation='relu'),
            Dropout(0.3),
            Dense(self.num_classes, activation='softmax')
        ])

        model.compile(optimizer='adam',
                     loss='categorical_crossentropy',
                     metrics=['accuracy'])
        return model

    def preprocess_chart_data(self, price_data, lookback_period=60):
        """
        Convert price time series to images for CNN input
        """
        images = []
        labels = []

        for i in range(lookback_period, len(price_data) - 1):
            # Create price chart image
            prices = price_data['close'].iloc[i-lookback_period:i]
            volumes = price_data['volume'].iloc[i-lookback_period:i]

            # Normalize prices
            price_min, price_max = prices.min(), prices.max()
            normalized_prices = (prices - price_min) / (price_max - price_min)

            # Create 2D image representation
            image = np.zeros((self.image_height, self.image_width))

            # Draw price line
            for j in range(len(normalized_prices) - 1):
                x1 = int(j * self.image_width / len(normalized_prices))
                y1 = int((1 - normalized_prices.iloc[j]) * self.image_height)
                x2 = int((j + 1) * self.image_width / len(normalized_prices))
                y2 = int((1 - normalized_prices.iloc[j + 1]) * self.image_height)

                # Simple line drawing
                if x1 < self.image_width and x2 < self.image_width:
                    for x in range(min(x1, x2), max(x1, x2) + 1):
                        if x1 != x2:
                            y = y1 + (y2 - y1) * (x - x1) / (x2 - x1)
                        else:
                            y = y1
                        if 0 <= int(y) < self.image_height:
                            image[int(y), x] = 1

            # Add volume information as background intensity
            vol_normalized = (volumes - volumes.min()) / (volumes.max() - volumes.min())
            for j in range(len(vol_normalized)):
                x = int(j * self.image_width / len(vol_normalized))
                intensity = vol_normalized.iloc[j] * 0.3  # Subtle background
                image[:, x] += intensity

            images.append(image.reshape(self.image_height, self.image_width, 1))

            # Create label based on future price movement
            future_return = (price_data['close'].iloc[i+1] - price_data['close'].iloc[i]) / price_data['close'].iloc[i]

            if future_return > 0.005:  # Up move
                label = [1, 0, 0]
            elif future_return < -0.005:  # Down move
                label = [0, 0, 1]
            else:  # Sideways
                label = [0, 1, 0]

            labels.append(label)

        return np.array(images), np.array(labels)

    def train(self, X_train, y_train, epochs=50, batch_size=32):
        """
        Train the CNN model
        """
        history = self.model.fit(
            X_train, y_train,
            epochs=epochs,
            batch_size=batch_size,
            validation_split=0.2,
            callbacks=[
                tf.keras.callbacks.EarlyStopping(patience=10, restore_best_weights=True),
                tf.keras.callbacks.ReduceLROnPlateau(patience=5, factor=0.5, min_lr=1e-6)
            ]
        )
        return history

    def predict_pattern(self, chart_image):
        """
        Predict price pattern from chart image
        """
        prediction = self.model.predict(chart_image.reshape(1, self.image_height, self.image_width, 1))
        class_names = ['UP', 'SIDEWAYS', 'DOWN']
        predicted_class = class_names[np.argmax(prediction[0])]
        confidence = np.max(prediction[0])

        return predicted_class, confidence

# Usage example
cnn_model = TechnicalPatternCNN()

# Load historical price data
price_data = pd.read_csv('eurusd_1h.csv', parse_dates=['timestamp'])
price_data = price_data.set_index('timestamp')

# Preprocess data into images
X, y = cnn_model.preprocess_chart_data(price_data)

# Train model
history = cnn_model.train(X, y, epochs=100)

# Make prediction on new chart
latest_prices = price_data.tail(60)
latest_image, _ = cnn_model.preprocess_chart_data(latest_prices)
prediction, confidence = cnn_model.predict_pattern(latest_image[0])

print(f"Predicted pattern: {prediction} (confidence: {confidence:.2%})")
```

### 2. Previsione di Prezzi con Deep Learning

#### 2.1 Long Short-Term Memory (LSTM) Networks

Le reti LSTM sono diventate lo standard per la previsione di serie temporali finanziarie grazie alla loro capacità di catturare dipendenze a lungo termine nei dati di mercato.

**Architettura LSTM per Previsione Prezzi:**

```python
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout, Bidirectional
from sklearn.preprocessing import MinMaxScaler
from sklearn.metrics import mean_squared_error, mean_absolute_error
import numpy as np

class PricePredictionLSTM:
    def __init__(self, sequence_length=60, n_features=5, n_predictions=1):
        self.sequence_length = sequence_length
        self.n_features = n_features
        self.n_predictions = n_predictions
        self.model = None
        self.scaler = MinMaxScaler()

    def build_model(self, units=128, dropout_rate=0.2):
        """
        Build LSTM model for price prediction
        """
        model = Sequential([
            # Bidirectional LSTM for capturing patterns in both directions
            Bidirectional(LSTM(units, return_sequences=True, input_shape=(self.sequence_length, self.n_features))),
            Dropout(dropout_rate),

            Bidirectional(LSTM(units//2, return_sequences=True)),
            Dropout(dropout_rate),

            Bidirectional(LSTM(units//4)),
            Dropout(dropout_rate),

            # Dense layers for prediction
            Dense(64, activation='relu'),
            Dropout(dropout_rate/2),
            Dense(self.n_predictions)
        ])

        model.compile(optimizer='adam', loss='mse', metrics=['mae'])
        self.model = model
        return model

    def prepare_sequences(self, data, target_column='close'):
        """
        Prepare input sequences for LSTM training
        """
        # Scale data
        scaled_data = self.scaler.fit_transform(data)

        X, y = [], []

        for i in range(self.sequence_length, len(scaled_data) - self.n_predictions + 1):
            # Input sequence
            X.append(scaled_data[i-self.sequence_length:i])

            # Target values
            target_idx = data.columns.get_loc(target_column)
            y.append(scaled_data[i:i+self.n_predictions, target_idx])

        return np.array(X), np.array(y)

    def create_features(self, price_data):
        """
        Create comprehensive feature set for price prediction
        """
        df = price_data.copy()

        # Price-based features
        df['returns'] = df['close'].pct_change()
        df['log_returns'] = np.log(df['close'] / df['close'].shift(1))

        # Moving averages
        for period in [5, 10, 20, 50]:
            df[f'ma_{period}'] = df['close'].rolling(period).mean()
            df[f'ma_{period}_slope'] = df[f'ma_{period}'].pct_change(5)

        # Volatility measures
        df['volatility_20'] = df['returns'].rolling(20).std()
        df['volatility_50'] = df['returns'].rolling(50).std()

        # Volume-based features
        df['volume_ma_20'] = df['volume'].rolling(20).mean()
        df['volume_ratio'] = df['volume'] / df['volume_ma_20']

        # Technical indicators
        # RSI
        delta = df['close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(14).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
        rs = gain / loss
        df['rsi'] = 100 - (100 / (1 + rs))

        # MACD
        ema_12 = df['close'].ewm(span=12).mean()
        ema_26 = df['close'].ewm(span=26).mean()
        df['macd'] = ema_12 - ema_26
        df['macd_signal'] = df['macd'].ewm(span=9).mean()
        df['macd_histogram'] = df['macd'] - df['macd_signal']

        # Bollinger Bands
        df['bb_middle'] = df['close'].rolling(20).mean()
        df['bb_std'] = df['close'].rolling(20).std()
        df['bb_upper'] = df['bb_middle'] + 2 * df['bb_std']
        df['bb_lower'] = df['bb_middle'] - 2 * df['bb_std']
        df['bb_position'] = (df['close'] - df['bb_lower']) / (df['bb_upper'] - df['bb_lower'])

        # Momentum indicators
        for period in [5, 10, 14]:
            df[f'momentum_{period}'] = df['close'] / df['close'].shift(period) - 1

        # Statistical features
        for period in [20, 50]:
            df[f'skew_{period}'] = df['returns'].rolling(period).skew()
            df[f'kurtosis_{period}'] = df['returns'].rolling(period).kurt()

        # Lagged features
        for lag in [1, 2, 3, 5, 10]:
            df[f'close_lag_{lag}'] = df['close'].shift(lag)
            df[f'returns_lag_{lag}'] = df['returns'].shift(lag)

        # Remove NaN values
        df = df.dropna()

        return df

    def train(self, X_train, y_train, epochs=100, batch_size=64, validation_split=0.2):
        """
        Train the LSTM model
        """
        callbacks = [
            tf.keras.callbacks.EarlyStopping(
                monitor='val_loss',
                patience=20,
                restore_best_weights=True
            ),
            tf.keras.callbacks.ReduceLROnPlateau(
                monitor='val_loss',
                factor=0.5,
                patience=10,
                min_lr=1e-6
            ),
            tf.keras.callbacks.ModelCheckpoint(
                'best_lstm_model.h5',
                monitor='val_loss',
                save_best_only=True
            )
        ]

        history = self.model.fit(
            X_train, y_train,
            epochs=epochs,
            batch_size=batch_size,
            validation_split=validation_split,
            callbacks=callbacks,
            verbose=1
        )

        return history

    def predict(self, X_input):
        """
        Make predictions with the trained model
        """
        predictions = self.model.predict(X_input)

        # Inverse transform predictions
        # Create dummy array with same shape as original scaled data
        dummy_array = np.zeros((len(predictions), self.n_features))
        dummy_array[:, 0] = predictions.flatten()  # Assuming close price is first column

        # Inverse transform
        unscaled_predictions = self.scaler.inverse_transform(dummy_array)[:, 0]

        return unscaled_predictions

    def evaluate_predictions(self, y_true, y_pred):
        """
        Comprehensive evaluation of predictions
        """
        # Scale back to original values for evaluation
        y_true_unscaled = self.scaler.inverse_transform(
            np.column_stack([y_true.flatten(), np.zeros((len(y_true), self.n_features-1))])
        )[:, 0]

        y_pred_unscaled = self.predict(
            np.column_stack([y_pred.flatten(), np.zeros((len(y_pred), self.n_features-1))])
        )

        # Calculate metrics
        mse = mean_squared_error(y_true_unscaled, y_pred_unscaled)
        rmse = np.sqrt(mse)
        mae = mean_absolute_error(y_true_unscaled, y_pred_unscaled)

        # Directional accuracy
        y_true_direction = np.sign(y_true_unscaled[1:] - y_true_unscaled[:-1])
        y_pred_direction = np.sign(y_pred_unscaled[1:] - y_pred_unscaled[:-1])
        directional_accuracy = np.mean(y_true_direction == y_pred_direction)

        # Sharpe ratio of prediction errors
        prediction_errors = y_true_unscaled - y_pred_unscaled
        sharpe_ratio = np.mean(prediction_errors) / np.std(prediction_errors) * np.sqrt(252)

        metrics = {
            'mse': mse,
            'rmse': rmse,
            'mae': mae,
            'directional_accuracy': directional_accuracy,
            'sharpe_ratio': sharpe_ratio
        }

        return metrics

# Usage example
lstm_predictor = PricePredictionLSTM(sequence_length=60, n_features=25)

# Load and prepare data
raw_data = pd.read_csv('sp500_daily.csv', parse_dates=['date'])
feature_data = lstm_predictor.create_features(raw_data)

# Prepare sequences
X, y = lstm_predictor.prepare_sequences(feature_data)

# Build and train model
lstm_predictor.build_model(units=256, dropout_rate=0.3)
history = lstm_predictor.train(X, y, epochs=200, batch_size=128)

# Make predictions
predictions = lstm_predictor.predict(X[-100:])  # Last 100 sequences

# Evaluate
metrics = lstm_predictor.evaluate_predictions(y[-100:], predictions.reshape(-1, 1))

print("Model Performance Metrics:")
for metric, value in metrics.items():
    print(f"{metric}: {value:.4f}")
```

### 3. Trading Algorithms con Reinforcement Learning

#### 3.1 Q-Learning per Strategie di Trading

Il Reinforcement Learning offre un framework potente per sviluppare strategie di trading che apprendono direttamente dall'interazione con il mercato.

**Implementazione di un Trading Agent con Deep Q-Learning:**

```python
import numpy as np
import pandas as pd
from collections import deque
import random
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout
from tensorflow.keras.optimizers import Adam

class TradingEnvironment:
    def __init__(self, price_data, initial_balance=100000, transaction_cost=0.001):
        self.price_data = price_data
        self.initial_balance = initial_balance
        self.transaction_cost = transaction_cost
        self.reset()

    def reset(self):
        self.current_step = 0
        self.balance = self.initial_balance
        self.position = 0  # 0: no position, 1: long, -1: short
        self.total_value = self.initial_balance
        self.trades = []
        return self._get_state()

    def _get_state(self):
        """
        Create state representation for the agent
        """
        if self.current_step >= len(self.price_data):
            return np.zeros(10)  # End of data

        current_data = self.price_data.iloc[self.current_step]

        # Technical indicators
        state = [
            current_data['close'] / 100,  # Normalized price
            current_data['returns'],      # Returns
            current_data['rsi'] / 100,    # RSI
            current_data['macd'],         # MACD
            current_data['bb_position'],  # Bollinger Band position
            self.position,                # Current position
            self.balance / self.initial_balance,  # Normalized balance
            self.total_value / self.initial_balance,  # Normalized total value
            len(self.trades) / 100,       # Number of trades (normalized)
            self.current_step / len(self.price_data)  # Time progress
        ]

        return np.array(state)

    def step(self, action):
        """
        Execute action and return reward
        Actions: 0=hold, 1=buy, 2=sell
        """
        current_price = self.price_data.iloc[self.current_step]['close']
        prev_value = self.total_value

        reward = 0
        done = False

        # Execute action
        if action == 1:  # Buy
            if self.position == 0:  # No position
                # Calculate position size (use portion of balance)
                position_value = self.balance * 0.1  # 10% of balance
                shares = position_value / current_price
                cost = shares * current_price * (1 + self.transaction_cost)

                if cost <= self.balance:
                    self.balance -= cost
                    self.position = 1
                    self.trades.append({
                        'step': self.current_step,
                        'action': 'buy',
                        'price': current_price,
                        'shares': shares
                    })
                    reward = -self.transaction_cost  # Small penalty for transaction

        elif action == 2:  # Sell
            if self.position == 1:  # Have long position
                # Find corresponding buy trade
                buy_trades = [t for t in self.trades if t['action'] == 'buy' and 'sell_price' not in t]
                if buy_trades:
                    buy_trade = buy_trades[-1]  # Last buy
                    shares = buy_trade['shares']
                    revenue = shares * current_price * (1 - self.transaction_cost)

                    self.balance += revenue
                    self.position = 0

                    # Record sell trade
                    buy_trade['sell_price'] = current_price
                    buy_trade['sell_step'] = self.current_step
                    buy_trade['pnl'] = revenue - (buy_trade['shares'] * buy_trade['price'] * (1 + self.transaction_cost))

                    reward = buy_trade['pnl'] / self.initial_balance  # Normalized P&L

        # Update total value
        if self.position == 1:
            # Find current position value
            buy_trades = [t for t in self.trades if t['action'] == 'buy' and 'sell_price' not in t]
            if buy_trades:
                current_position_value = buy_trades[-1]['shares'] * current_price
                self.total_value = self.balance + current_position_value
            else:
                self.total_value = self.balance
        else:
            self.total_value = self.balance

        # Move to next step
        self.current_step += 1
        if self.current_step >= len(self.price_data) - 1:
            done = True

        # Additional reward based on portfolio change
        portfolio_change = (self.total_value - prev_value) / self.initial_balance
        reward += portfolio_change

        next_state = self._get_state()

        return next_state, reward, done, {}

class DQTradingAgent:
    def __init__(self, state_size, action_size):
        self.state_size = state_size
        self.action_size = action_size
        self.memory = deque(maxlen=2000)
        self.gamma = 0.95    # discount rate
        self.epsilon = 1.0   # exploration rate
        self.epsilon_min = 0.01
        self.epsilon_decay = 0.995
        self.learning_rate = 0.001
        self.model = self._build_model()
        self.target_model = self._build_model()
        self.update_target_model()

    def _build_model(self):
        """Neural Net for Deep-Q learning"""
        model = Sequential()
        model.add(Dense(64, input_dim=self.state_size, activation='relu'))
        model.add(Dropout(0.2))
        model.add(Dense(64, activation='relu'))
        model.add(Dropout(0.2))
        model.add(Dense(32, activation='relu'))
        model.add(Dense(self.action_size, activation='linear'))
        model.compile(loss='mse', optimizer=Adam(lr=self.learning_rate))
        return model

    def update_target_model(self):
        """Copy weights from model to target_model"""
        self.target_model.set_weights(self.model.get_weights())

    def remember(self, state, action, reward, next_state, done):
        self.memory.append((state, action, reward, next_state, done))

    def act(self, state):
        if np.random.rand() <= self.epsilon:
            return random.randrange(self.action_size)
        act_values = self.model.predict(state.reshape(1, -1), verbose=0)
        return np.argmax(act_values[0])

    def replay(self, batch_size):
        minibatch = random.sample(self.memory, batch_size)
        for state, action, reward, next_state, done in minibatch:
            target = reward
            if not done:
                target = (reward + self.gamma *
                         np.amax(self.target_model.predict(next_state.reshape(1, -1), verbose=0)[0]))
            target_f = self.model.predict(state.reshape(1, -1), verbose=0)
            target_f[0][action] = target
            self.model.fit(state.reshape(1, -1), target_f, epochs=1, verbose=0)
        if self.epsilon > self.epsilon_min:
            self.epsilon *= self.epsilon_decay

    def load(self, name):
        self.model.load_weights(name)

    def save(self, name):
        self.model.save_weights(name)

# Training the trading agent
def train_trading_agent(price_data, episodes=1000, batch_size=32):
    env = TradingEnvironment(price_data)
    state_size = 10  # State features
    action_size = 3  # Actions: hold, buy, sell
    agent = DQTradingAgent(state_size, action_size)

    for e in range(episodes):
        state = env.reset()
        total_reward = 0

        for time in range(500):  # Max steps per episode
            action = agent.act(state)
            next_state, reward, done, _ = env.step(action)

            agent.remember(state, action, reward, next_state, done)
            state = next_state
            total_reward += reward

            if done:
                agent.update_target_model()
                print(f"Episode: {e}/{episodes}, Total Reward: {total_reward:.4f}, Epsilon: {agent.epsilon:.4f}")
                break

        if len(agent.memory) > batch_size:
            agent.replay(batch_size)

    return agent, env

# Usage
# Load price data with technical indicators
price_data = pd.read_csv('trading_data.csv')
price_data = add_technical_indicators(price_data)

# Train agent
trained_agent, env = train_trading_agent(price_data, episodes=2000)

# Evaluate performance
test_data = price_data.tail(500)
env_test = TradingEnvironment(test_data)

state = env_test.reset()
total_reward = 0
portfolio_values = [env_test.total_value]

for _ in range(len(test_data) - 1):
    action = trained_agent.act(state)
    next_state, reward, done, _ = env_test.step(action)
    state = next_state
    total_reward += reward
    portfolio_values.append(env_test.total_value)

    if done:
        break

# Calculate performance metrics
returns = pd.Series(portfolio_values).pct_change().dropna()
sharpe_ratio = returns.mean() / returns.std() * np.sqrt(252)
max_drawdown = (pd.Series(portfolio_values).cummax() - pd.Series(portfolio_values)).max() / pd.Series(portfolio_values).cummax().max()

print(f"Test Sharpe Ratio: {sharpe_ratio:.4f}")
print(f"Max Drawdown: {max_drawdown:.4%}")
print(f"Total Return: {(portfolio_values[-1] / portfolio_values[0] - 1):.2%}")
```
